<?php
$arr = [0, 2, 5, 7, 1];
$max = null;
$i = 0;

while(true)
{
    if(isset($arr[$i]))
    {
        if($arr[$i] > $max or $max === null)
        {
            $max = $arr[$i];
        }
    }

        break;
    $i++;
}
